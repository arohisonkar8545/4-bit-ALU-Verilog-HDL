module alu(
    input [3:0] a,
    input [3:0] b,
    input [1:0] sel,
    output reg [3:0] y
);

always @(*) begin
    case(sel)
        2'b00: y = a + b;   // Addition
        2'b01: y = a - b;   // Subtraction
        2'b10: y = a & b;   // AND
        2'b11: y = a | b;   // OR
        default: y = 4'b0000;
    endcase
end

endmodule
